  def look(nick, name)
    temp = @user_list[nick].place
    res = nil
    @place_list[temp].people.each do |p|
      if (p.class == Npc and p.name == name)
        res = p
        break
      end
    end
    return "#{res.name}, #{res.descr}" if res
    # se nn e' un npc controlla gli oggetti con quel nome ecc
    # da fare ...
    return get_text(:nothing)
  end
