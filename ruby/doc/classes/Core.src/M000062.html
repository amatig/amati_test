  def users_zone(nick)
    me = @user_list[nick]
    u = []
    @place_list[me.place].people.each do |p|
      if p.class == User
        u << bold(p.name) if (p != me)
      else
        u << italic(p.name)
      end
    end
    if u.empty?
      c = get_text(:nobody) + ","
      u = [get_text(:onlyu)]
    else
      c = get_text((u.length > 1) ? :ci_sono : :c_e)
    end
    return get_text(:uz) % [c, list(u)]
  end
