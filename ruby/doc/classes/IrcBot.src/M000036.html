  def main_loop()
    while true
      return if @irc.eof
      msg = @irc.gets
      Thread.new do
        msg = msg.strip
        if msg =~ /^PING :(.+)$/i
          puts "[ Server ping ]"
          send "PONG :#{$1}"
        else
          parse msg
        end
      end
    end
  end
