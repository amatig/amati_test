  def connectIRC(server = "127.0.0.1", port = 6667)
    @irc = TCPSocket.open(server, port)
    temp = "USER #{@nick} #{@nick} bla :#{@realname}#{@delim}"
    temp += "NICK #{@nick}#{@delim}"
    @irc.send(temp, 0)
    
    # buffer send loop
    Thread.new do
      while true do
        socket_send
        sleep 1
      end
    end
  end
